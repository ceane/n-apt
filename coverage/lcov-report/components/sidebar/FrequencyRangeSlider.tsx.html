              width: `max(${widthPercent}%, ${minContentThumbWidth}px)`,
            }}
            onMouseDown={handleMouseDown}
          >
            <WindowLabel ref={windowLabelRef} $isActive={isActive}>
              {formatFreq(currentMin)} - {formatFreq(currentMax)}
            </WindowLabel>
          </VisibleWindow>
        </RangeTrack>
      </SliderContainer>
    </SliderWrapper>
  );
};
 
export default FrequencyRangeSlider;
